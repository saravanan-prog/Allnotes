17.Query Operators  =
                         
       17.1 Comparison Operator
        ------------------------
        
                        
        $eq  (==)               :   Values are equal
                                            
                                            
                                    db.departments.find({
                                        departmentName: {
                                            $eq:"Commerce and Accountancy"
                                            }
                                        }
                                    )




        $ne  (!=)              :   Values are not equal


                                        db.departments.find({
                                            departmentName: {
                                                $ne:"Commerce and Accountancy"
                                                }
                                            }
                                        )




        $gt  (>)               :   Value is greater than another value
                                        
                                    db.departments.find({
                                        staffCount: {
                                            $gt:100
                                            }
                                        }
                                    )


        $gte (>=)              :   Value is greater than or equal to another value

                                        db.departments.find({
                                            staffCount: {
                                                $gte:100
                                                }
                                            }
                                        )


        $lt  (<)               :   Value is less than another value

      

                                        db.departments.find({
                                            staffCount: {
                                                $lt:2
                                                }
                                            }
                                        )




        $lte (<=)              :   Value is less than or equal to another value
                                        
                                        db.departments.find({
                                            staffCount: {
                                                $lte:2
                                                }
                                            }
                                        )

                                            
        $in  (IN)              :   Value is matched within an array

                                        db.departments.find({
                                            staffCount: {
                                                $in:['Computer Application','Commerce']
                                                }
                                            }
                                        )



        17.2 Logical
        ------------
                  
        $and:  ( AND )      Returns documents where both queries match    

                Syntax:
                -------
                    { $and: [ { Expression1 }, { Expression2 }, ..., { ExpressionN } ] }
                       db.courses.find(
                            {
                                $and:[ 
                                { courseID:'JD106'},
                                { CourseName:'Credo comobo pack'}
                                ]
                            }
                        )


        $or:    ( OR )      Returns documents where either query matches
        $nor:               Returns documents where both queries fail to match
        $not:   (!)         Returns documents where the query does not match
