CREATE TABLE ADDRESS(
	ID INT PRIMARY KEY auto_increment,
    TEMP_ADDRESS VARCHAR(20),
    PER_ADDRESS VARCHAR(20),
    STUDENT_ID INT,
	foreign key (STUDENT_ID) REFERENCES STUDENTS(ID)
    ON DELETE cascade
